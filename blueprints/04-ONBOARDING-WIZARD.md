# Blueprint 04: Onboarding Wizard

> First-run experience that takes users from zero to functional Helix

## Design Principles

1. **Smart Defaults**: Minimize decisions for normies, advanced options available
2. **Progressive Disclosure**: Start simple, reveal complexity when needed
3. **No Dead Ends**: Every step has a way forward, even if skipped
4. **Immediate Value**: Show Helix working as soon as possible
5. **Educational**: Explain *why* things matter, not just *what*

## State Management

### Wizard State Schema

```typescript
interface OnboardingState {
  // Progress
  currentStep: number;
  completedSteps: Set<string>;
  isAdvancedMode: boolean;

  // Account
  account: {
    connected: boolean;
    provider?: 'email' | 'github' | 'google';
    userId?: string;
    email?: string;
  };

  // AI Provider
  aiProvider: {
    primary: {
      provider: 'anthropic' | 'openai' | 'google' | 'bedrock' | 'local';
      apiKey?: string; // Stored in keyring, not state
      model: string;
      validated: boolean;
    };
    fallbacks: Array<{
      provider: string;
      configured: boolean;
    }>;
  };

  // Discord Logging
  discord: {
    enabled: boolean;
    webhooks: {
      commands?: string;
      api?: string;
      heartbeat?: string;
      consciousness?: string;
      alerts?: string;
    };
    validated: boolean;
  };

  // Personality
  personality: {
    communicationStyle: {
      verbosity: number; // 0-100
      formality: number; // 0-100
      technicality: number; // 0-100
    };
    preferences: {
      askBeforeActing: boolean;
      shareThinking: boolean;
    };
    useCases: string[];
    // Advanced
    enneagram?: string;
    coreValues?: string[];
    userName?: string;
    initialTrust?: number;
  };

  // Privacy
  privacy: {
    telemetryEnabled: boolean;
    tier: 'free' | 'phantom' | 'overseer' | 'architect';
  };

  // Channels
  channels: {
    whatsapp: { configured: boolean; };
    discord: { configured: boolean; serverId?: string; };
    slack: { configured: boolean; workspaceId?: string; };
    telegram: { configured: boolean; botToken?: string; };
  };

  // First Chat
  firstChat: {
    completed: boolean;
    messages: Array<{
      role: 'user' | 'assistant';
      content: string;
    }>;
  };
}
```

### Persistence

```typescript
// Save wizard state to disk (for resume on crash)
async function saveWizardState(state: OnboardingState): Promise<void> {
  const statePath = await join(await helixDir(), 'onboarding-state.json');

  // Never persist sensitive data
  const safeState = {
    ...state,
    aiProvider: {
      ...state.aiProvider,
      primary: {
        ...state.aiProvider.primary,
        apiKey: undefined, // Stored in keyring
      },
    },
    discord: {
      ...state.discord,
      webhooks: undefined, // Stored encrypted
    },
  };

  await writeTextFile(statePath, JSON.stringify(safeState, null, 2));
}

// Load wizard state (resume interrupted setup)
async function loadWizardState(): Promise<OnboardingState | null> {
  const statePath = await join(await helixDir(), 'onboarding-state.json');

  try {
    const content = await readTextFile(statePath);
    return JSON.parse(content);
  } catch {
    return null;
  }
}
```

## Component Implementation

### WizardContainer.tsx

```typescript
import React, { useState, useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import { invoke } from '@tauri-apps/api/core';
import { WelcomeStep } from './WelcomeStep';
import { AccountStep } from './AccountStep';
import { ApiKeyStep } from './ApiKeyStep';
import { DiscordStep } from './DiscordStep';
import { PersonalityStep } from './PersonalityStep';
import { PrivacyStep } from './PrivacyStep';
import { FirstChatStep } from './FirstChatStep';
import { CompletionStep } from './CompletionStep';
import { useOnboarding } from '@/hooks/useOnboarding';

const STEPS = [
  { id: 'welcome', component: WelcomeStep, required: true },
  { id: 'account', component: AccountStep, required: false },
  { id: 'api-key', component: ApiKeyStep, required: true },
  { id: 'discord', component: DiscordStep, required: false },
  { id: 'personality', component: PersonalityStep, required: true },
  { id: 'privacy', component: PrivacyStep, required: true },
  { id: 'first-chat', component: FirstChatStep, required: false },
  { id: 'completion', component: CompletionStep, required: true },
];

export function WizardContainer() {
  const navigate = useNavigate();
  const {
    state,
    updateState,
    currentStep,
    setCurrentStep,
    canProceed,
    isAdvancedMode,
    setAdvancedMode,
  } = useOnboarding();

  const [isLoading, setIsLoading] = useState(true);

  // Check if already onboarded
  useEffect(() => {
    async function checkOnboarded() {
      const onboarded = await invoke<boolean>('is_first_run');
      if (!onboarded) {
        navigate('/chat');
        return;
      }
      setIsLoading(false);
    }
    checkOnboarded();
  }, [navigate]);

  const handleNext = async () => {
    // Find next required step or just next step
    let nextIndex = currentStep + 1;

    if (!isAdvancedMode) {
      // Skip optional steps in fast mode
      while (nextIndex < STEPS.length && !STEPS[nextIndex].required) {
        nextIndex++;
      }
    }

    if (nextIndex >= STEPS.length) {
      // Complete onboarding
      await finishOnboarding();
      return;
    }

    setCurrentStep(nextIndex);
  };

  const handleBack = () => {
    let prevIndex = currentStep - 1;

    if (!isAdvancedMode) {
      // Skip optional steps in fast mode
      while (prevIndex >= 0 && !STEPS[prevIndex].required) {
        prevIndex--;
      }
    }

    if (prevIndex >= 0) {
      setCurrentStep(prevIndex);
    }
  };

  const handleSkip = () => {
    handleNext();
  };

  const finishOnboarding = async () => {
    // Save final configuration
    await invoke('save_onboarding_config', { state });

    // Mark as onboarded
    await invoke('mark_onboarded');

    // Navigate to main app
    navigate('/chat');
  };

  if (isLoading) {
    return (
      <div className="flex items-center justify-center h-screen">
        <div className="animate-pulse">Loading...</div>
      </div>
    );
  }

  const CurrentStepComponent = STEPS[currentStep].component;
  const currentStepConfig = STEPS[currentStep];

  return (
    <div className="min-h-screen bg-void-950 flex flex-col">
      {/* Progress indicator */}
      <div className="p-4 flex justify-center gap-2">
        {STEPS.filter(s => isAdvancedMode || s.required).map((step, index) => (
          <div
            key={step.id}
            className={`w-2 h-2 rounded-full transition-colors ${
              index === currentStep
                ? 'bg-blue-500'
                : index < currentStep
                ? 'bg-blue-500/50'
                : 'bg-gray-700'
            }`}
          />
        ))}
      </div>

      {/* Step content */}
      <div className="flex-1 flex items-center justify-center p-8">
        <div className="w-full max-w-2xl">
          <CurrentStepComponent
            state={state}
            updateState={updateState}
            onNext={handleNext}
            onBack={handleBack}
            onSkip={handleSkip}
            isRequired={currentStepConfig.required}
            canProceed={canProceed(currentStepConfig.id)}
            isAdvancedMode={isAdvancedMode}
            setAdvancedMode={setAdvancedMode}
          />
        </div>
      </div>

      {/* Navigation */}
      <div className="p-4 flex justify-between items-center border-t border-gray-800">
        <button
          onClick={handleBack}
          disabled={currentStep === 0}
          className="px-4 py-2 text-gray-400 hover:text-white disabled:opacity-50"
        >
          â† Back
        </button>

        <div className="flex gap-4">
          {!currentStepConfig.required && (
            <button
              onClick={handleSkip}
              className="px-4 py-2 text-gray-400 hover:text-white"
            >
              Skip for now
            </button>
          )}

          <button
            onClick={handleNext}
            disabled={!canProceed(currentStepConfig.id)}
            className="px-6 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-500 disabled:opacity-50 disabled:cursor-not-allowed"
          >
            {currentStep === STEPS.length - 1 ? 'Finish' : 'Continue â†’'}
          </button>
        </div>
      </div>
    </div>
  );
}
```

## Validation Rules

| Step | Required Fields | Validation |
|------|-----------------|------------|
| Welcome | None | Always valid |
| Account | None (optional) | OAuth flow must complete if started |
| API Key | provider, apiKey | Must pass API test call |
| Discord | webhooks (if enabled) | Each webhook must respond to test |
| Personality | communicationStyle | All sliders must have values |
| Privacy | telemetryEnabled | Must make explicit choice |
| First Chat | None (optional) | None |
| Completion | None | All required steps completed |
